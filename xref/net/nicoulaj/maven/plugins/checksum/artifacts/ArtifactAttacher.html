17  package net.nicoulaj.maven.plugins.checksum.artifacts;
18  
19  import org.apache.maven.project.MavenProject;
20  import org.apache.maven.project.MavenProjectHelper;
21  
22  import java.io.File;
23  
24  public class ArtifactAttacher implements ArtifactListener {
25      private final MavenProject project;
26      private final MavenProjectHelper projectHelper;
27  
28      public ArtifactAttacher(MavenProject project, MavenProjectHelper projectHelper) {
29          this.project = project;
30          this.projectHelper = projectHelper;
31      }
32  
33      @Override
34      public void artifactCreated(File artifact, String checksumType, String artifactType, String artifactClassifier) {
35          if (checksumType.startsWith(".")) {
36              // Project helper expects a type without leading dot (e.g. turn ".md5" into "md5").
37              checksumType = checksumType.substring(1);
38          }
39  
40          String attachmentType = artifactType == null ? checksumType : artifactType + "." + checksumType;
41  
42          projectHelper.attachArtifact(project, attachmentType, artifactClassifier, artifact);
43      }
44  }
