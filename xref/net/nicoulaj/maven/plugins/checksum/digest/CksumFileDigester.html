42  	public CksumFileDigester()
43  	{
44  		super(ALGORITHM);
45  	}

47  	@Override
48  	public String calculate(File file) throws DigesterException
49  	{
50  		int value = 0;
51  		long length = 0L;
52  
53  		try (BufferedInputStream fis = new BufferedInputStream( new FileInputStream(file) ))
54  		{
55  			int byteRead;
56  			while ( (byteRead = fis.read()) >= 0 )
57  			{
58  				value = value << 8 ^ CRC_TAB[(value >> 24 ^ (byte)(byteRead & 0xFF)) & 0xFF];
59  				length += 1L;
60  			}
61  		    while ( length != 0L )
62  		    {
63  		    	value = value << 8 ^ CRC_TAB[(value >> 24 ^ (int)(length & 0xFF)) & 0xFF];
64  		    	length >>= 8;
65  		    }
66  		    return Long.toString((value ^ 0xFFFFFFFFL) & 0xFFFFFFFFL);
67  		}
68          catch ( FileNotFoundException e )
69          {
70              throw new DigesterException( "Unable to read " + file.getPath() + ": " + e.getMessage() );
71          }
72  		catch ( IOException e )
73  		{
74  			throw new DigesterException(
75  	                "Unable to calculate the " + getAlgorithm() + " hashcode for " + file.getPath() + ": "
76  	                    + e.getMessage() );
77  		}
78  	}
