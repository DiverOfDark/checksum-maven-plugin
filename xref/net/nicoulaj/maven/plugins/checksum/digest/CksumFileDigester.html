35  	@Override
36  	public String calculate(File file) throws DigesterException
37  	{
38  		int value = 0;
39  		long length = 0L;
40  		BufferedInputStream fis;
41  
42  		try
43  		{
44  			fis = new BufferedInputStream( new FileInputStream(file) );
45  		}
46  		catch ( FileNotFoundException e )
47  		{
48  			throw new DigesterException( "Unable to read " + file.getPath() + ": " + e.getMessage() );
49  		}
50  
51  		try
52  		{
53  			int byteRead;
54  			while ( (byteRead = fis.read()) >= 0 )
55  			{
56  				value = value << 8 ^ CRC_TAB[(value >> 24 ^ (byte)(byteRead & 0xFF)) & 0xFF];
57  				length += 1L;
58  			}
59  		    while ( length != 0L )
60  		    {
61  		    	value = value << 8 ^ CRC_TAB[(value >> 24 ^ (int)(length & 0xFF)) & 0xFF];
62  		    	length >>= 8;
63  		    }
64  		    return Long.toString((value ^ 0xFFFFFFFFL) & 0xFFFFFFFFL);
65  		}
66  		catch ( IOException e )
67  		{
68  			throw new DigesterException(
69  	                "Unable to calculate the " + getAlgorithm() + " hashcode for " + file.getPath() + ": "
70  	                    + e.getMessage() );
71  		}
72  		finally
73  		{
74  			if (fis != null)
75  			{
76  				try
77  				{
78  					fis.close();
79  				}
80  				catch (IOException e)
81  				{
82  					throw new DigesterException(
83  			                "Unable to calculate the " + getAlgorithm() + " hashcode for " + file.getPath() + ": "
84  			                    + e.getMessage() );
85  				}
86  			}
87  		}
88  	}
